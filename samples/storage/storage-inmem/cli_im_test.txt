# commands to create, open, etc a postgres wallet in CLI
# run cli as:
# RUST_BACKTRACE=1 LD_LIBRARY_PATH=../samples/storage/storage-inmem/target/debug/ cargo run ../samples/storage/storage-inmem/cli_im_test.txt
# register the postgres plug-in
wallet register inmem_custom so_file=libindystrginmem.dylib prefix=inmemwallet_fn_
# create a inmem wallet
wallet create wallet_imx key storage_type=inmem_custom 
# open the inmem_custom wallet
wallet open wallet_imx key 
# create and use a new DID in the wallet
did new seed=00000000000000000000000000000My5 metadata=my_did_metadata
did use VREsKVpdEtWCcCmMfddYuT
# close and then delete the wallet
wallet close
wallet delete wallet_imx key
